# Eksempel: sæt virksomhedens betalingsoplysninger efter `init`.
# Sprang du bankkontoen over ved `init`, sætter denne kommando den — så
# fremtidige fakturaer får en BETALING-blok på PDF'en (reg.nr./kontonr./IBAN).
# Kun de flag du faktisk angiver, ændres; resten beholder deres værdi.
rentemester company set-profile --company <sti-eller-slug> \
  --bank-name "Danske Bank" \
  --bank-reg 1234 \
  --bank-account 0001234567 \
  --iban DK5000400440116243
